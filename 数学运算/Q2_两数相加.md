给你两个 非空 的链表，表示两个非负的整数。它们每位数字都是按照 逆序 的方式存储的，并且每个节点只能存储 一位 数字。

请你将两个数相加，并以相同形式返回一个表示和的链表。

你可以假设除了数字 0 之外，这两个数都不会以 0 开头。



思路：使用加法原则，逐位遍历两个链表并相加，记录进位，用于之后的计算中

```c++
ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {     
        int carry = 0;
        ListNode *h = new ListNode(0);
        ListNode *head = h;

        //同时遍历两链表数字，并使用加法相加，保留进位
        while(l1 != NULL  && l2 != NULL){
            int num = l1->val + l2->val + carry;
            ListNode *n = new ListNode(0);
            carry = 0; //用掉进位清0 
            if( num >= 10 ){
                n->val = num - 10;
                carry += 1;
            } 
            else
                n->val = num;

            h->next = n;
            h = h->next;
            l1 = l1->next;
            l2 = l2->next;
        }

        while(l1 != NULL){
            int num = l1->val + carry;
            ListNode *n = new ListNode(0);
            carry = 0;

            if( num >= 10 ){
                n->val = num - 10;
                carry += 1;
            } 
            else
                n->val = num;

            h->next = n;
            h = h->next;
            l1 = l1->next;
        }

        while(l2 != NULL){
            int num = l2->val + carry;
            ListNode *n = new ListNode(0);
            carry = 0;

            if( num >= 10 ){
                n->val = num - 10;
                carry += 1;
            } 
            else
                n->val = num;

            h->next = n;
            h = h->next;
            l2 = l2->next;
        }

        if(carry == 1){ //计算结束后进位仍存在
            ListNode *n = new ListNode(1);
            h->next = n;
            h = h->next;
        }
        
        return head->next;
    }
```

代码优化：

```c++
ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {     
        ListNode *head = nullptr, *tail = nullptr;
        int carry = 0;
        while (l1 || l2) {
            int n1 = l1 ? l1->val: 0;
            int n2 = l2 ? l2->val: 0;
            int sum = n1 + n2 + carry;
            if (!head) {
                head = tail = new ListNode(sum % 10);
            } else {
                tail->next = new ListNode(sum % 10);
                tail = tail->next;
            }
            carry = sum / 10;
            if (l1) {
                l1 = l1->next;
            }
            if (l2) {
                l2 = l2->next;
            }
        }
        if (carry > 0) {
            tail->next = new ListNode(carry);
        }
        return head;
    }
```
